      <button class="calculator-key" type="button" onclick="calculate()">=</button>
    </div>

  <script>
    function clearScreen() {
      document.getElementById('calculator-screen').innerHTML = ''
    }

    function calculate() {
      const expression = document.getElementById('calculator-screen').innerHTML
      const result = eval(expression)
      document.getElementById('calculator-screen').innerHTML = result
    }

    const keys = document.querySelectorAll('.calculator-key')
    keys.forEach(function (key) {
      key.addEventListener('click', function () {
        const value = this.value
        document.getElementById('calculator-screen').innerHTML += value
      })
    })
  </script>
